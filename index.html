Firebase-like API to access CouchDB data in real time that support offline data sync

Simple API access

Access your data with a single line of code

var Chats = new Cippy('http://localhost:5984/chats');

Three-way data binding

Keep your data always in sync

Chats.bindTo($scope, 'chats');

CRUD data

Create, read, update, delete your data

Chats.add({ user: 'sonny', message: 'Hello World' });

Simple Chat

This example shows how easy it is to implement chatting app.

<!DOCTYPE html>
<html>
  <body ng-app="SimpleChat">
    <div ng-controller="AppCtrl">
      <form>
        <input type="text" ng-model="name" placeholder="Name">
        <textarea ng-model="message" placeholder="Your Message..."></textarea>
        <button ng-click="send()">Send</button>
      </form>
      <div>
        <div ng-repeat="message in chats">
          <strong>{{message.name}} : </strong>
          {{message.message}}
        </div>
      </div>
    </div>
    <script src="vendor/js/underscore-min.js"></script>
    <script src="vendor/js/angular.min.js"></script>
    <script src="vendor/js/pouchdb.js"></script>
    <script src="src/cippy.js"></script>
    <script>
      angular.module('SimpleChat', ['cippydb'])
      .controller('AppCtrl', function($scope, Cippy) {
        $scope.chats = [];
        $scope.name = '';
        $scope.message = '';
        var ChatDB = new Cippy('http://localhost:5984/chats');
        ChatDB.bindTo($scope, 'chats');

        $scope.send = function() {
          ChatDB.add({
            _id: new Date().toISOString(),
            name: $scope.name,
            message: $scope.message
          });
          $scope.message = '';
        }
      });
    </script>
  </body>
</html>
